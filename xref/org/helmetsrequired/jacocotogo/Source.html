245     /**
246      * Validates that valid input parameters are specified.
247      */
248     public void validate() {
249         if (serviceURL == null) {
250             // type is required
251             if (type == null) {
252                 throw new IllegalArgumentException("Parameter 'type' is missing.  It is required if 'serviceURL' is not set.");
253             }
254             try {                
255                 sourceType = SourceType.valueOf(type.toUpperCase());
256             } catch (IllegalArgumentException ex) {                            
257                 throw new IllegalArgumentException("Parameter 'type' has invalid value: '" + type + "' valid values are: '" + SourceType.values().toString(), ex);
258             }
259             if (hostname == null) {
260                 throw new IllegalArgumentException("Parameter 'hostname' is missing.  It is required if 'serviceURL' is not set.");
261             }
262             validateHostname();
263             validatePort();
264             if (sourceType == SourceType.JMX) {
265                 constructJMXServiceURL();
266             }
267         } else {
268             parseServiceURL();
269         }
270

273     private InetAddress validateHostname() {
274         if (hostname == null || hostname.trim().isEmpty()) {
275             throw new IllegalArgumentException("Parameter 'hostname' is not provided.");
276         }
277         logger.debug("Verifying that hostname: '{}' can be resolved.", hostname);
278         try {
279             return InetAddress.getByName(hostname);
280         } catch (UnknownHostException ex) {
281             throw new RuntimeException("Could not resolve hostname: '" + hostname + "'.", ex);
282         }
283     }

291     private void constructJMXServiceURL() {
292         serviceURL = DEFAULT_JMX_URL_PREFIX + hostname + ":" + port + DEFAULT_JMX_URL_SUFFIX;
293     }
294 
295     private void parseServiceURL() {
296         if (serviceURL == null) {
297             throw new IllegalArgumentException("Null 'serviceURL'.  It is required if 'type', 'hostname', and 'port' are not set.");
298         }
299         String[] tokens = serviceURL.split(":");
300         if (tokens.length < 3) {
301             throw new IllegalArgumentException("Invalid 'serviceURL'.");
302         }
303         if (tokens[0].equalsIgnoreCase("tcp")) {
304             parseTCPServiceURL(tokens);
305         } else if (tokens[0].equals("service") && tokens[1].equals("jmx")) {
306             parseJMXServiceURL(tokens);
307         } else {
308             throw new IllegalArgumentException("Invalid 'serviceURL'.  Expected 'serviceURL' to start with 'tcp' or 'service:jmx'");
309         }
310     }
311 
312     private void parseTCPServiceURL(String[] tokens) {
313         if (tokens.length > 3) {
314             throw new IllegalArgumentException("Invalid 'serviceURL'.  For tcp 'serviceURL' should be in the format of: 'tcp://<hostname>:<portno>'.");
315         }
316         sourceType = SourceType.TCP;
317         if (!tokens[1].startsWith("//")) {
318             throw new IllegalArgumentException("Invalid 'serviceURL'.  For tcp 'serviceURL' should be in the format of: 'tcp://<hostname>:<portno>'.");
319         }
320         hostname = tokens[1].substring(2);
321         validateHostname();
322         port = Integer.parseInt(tokens[2]);
323         validatePort();
324     }
325 
326     private void parseJMXServiceURL(String[] tokens) {
327         sourceType = SourceType.JMX;
328     }
